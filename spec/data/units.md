## UnitDef TypeScript interface

```typescript
export interface UnitDef {
  id: string;
  name: string;
  class: UnitClass;           // 'infantry'|'archer'|'cavalry'|'siege'|'monk'|'villager'|'ship'
  hp: number;
  speed: number;              // tiles/second
  attackDamage: number;
  attackRange: number;        // 0 = melee
  attackSpeed: number;        // attacks/second
  attackType: AttackType;     // 'melee'|'ranged'|'siege'|'special'
  armor: { melee: number; pierce: number };
  lineOfSight: number;
  cost: { food: number; wood: number; gold: number; stone: number };
  trainTime: number;          // seconds
  populationCost: number;
  minAge: UnitAge;            // 'dark'|'feudal'|'castle'|'imperial'
  upgradesTo?: string;        // id of upgraded form
  uniqueToCivId?: string;
}
```
